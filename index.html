        var Minimal_Angle = (h, m) => 
        {
          let angle = Math.abs((30 * h) - (11 / 2) * m);
          if (angle > 180) {
            angle = 360 - angle;
          }
          return angle;      
        };

Perfect Number Check. 

        var Perfect_Check = (N) => 
        {
           
        
          let sum = 0;
          for (let i = 1; i < (N/2); i++) {
            if (N % i === 0) {
              sum += i;
            }
          }
          if(sum === N){
            return 'YES';
          }
        else{
          return 'NO';
        }  
                
        };

    var Reverse_Number = (N) => 
    {   
     let rev = 0;

        while (N > 0) {
        let rem = N % 10;
         rev = rev * 10 + rem;
        N = Math.floor(N / 10);
    }
        return rev;
    };
